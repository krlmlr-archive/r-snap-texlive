                [The MLA Style Manual]

  \let\AttribInit\sffamily
  \let\PreTrib\textsection
  \let\PostTrib\ShortEmpty
  \lessoninternal
  } {%
  \endlessoninternal
  }

  \RestoreDoXVarS
}
