     Als erstes eine Methode:

     \begin{method}
      \head{int div(int a, int b, double \&c);}
      \para{a}{Dividend}
      \para{b}{Divisor}
      \para{\&c}{Ergebnis der Division}
      \precond{keine}
      \descr{Dividiert \texttt{a} durch \texttt{b} und legt
             das Ergebnis in \texttt{c} ab.}
      \postcond{keine}
      \error{keine}
      \return{\texttt{-1}, falls \texttt{b==0}, sonst \texttt{0}}
      \see{Duden der Mathematik}
     \end{method}
